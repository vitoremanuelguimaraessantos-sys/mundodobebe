                if (existingItem.qty + qty <= product.stock) {
                    existingItem.qty += qty;
                    showToast(`Quantidade atualizada! (${existingItem.qty}x ${size})`);
                } else {
                    showToast('Stock máximo atingido para este tamanho!');
                    return;
                }
            } else {
                cart.push({
                    id: product.id,
                    name: product.name,
                    price: product.salePrice || product.price,
                    image: product.image,
                    size: size,
                    qty: qty
                });
                showToast(`Produto adicionado! (${qty}x ${size})`);
            }
            updateCartUI();
        }

        function removeFromCart(id, size) {
            cart = cart.filter(item => !(item.id === id && item.size === size));
            updateCartUI();
        }

        function updateCartQty(id, size, change) {
            const item = cart.find(item => item.id === id && item.size === size);
            const product = products.find(p => p.id === id);
            
            if (item) {
                const newQty = item.qty + change;
                if (newQty > 0 && newQty <= product.stock) {
                    item.qty = newQty;
                } else if (newQty <= 0) {
                    removeFromCart(id, size);
                    return;
                }
                updateCartUI();
            }
        }
